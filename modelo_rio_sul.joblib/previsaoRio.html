def evaluate_model(y_true, y_pred):
    rmse = np.sqrt(mean_squared_error(y_true, y_pred))
    mae = mean_absolute_error(y_true, y_pred)
    r2 = r2_score(y_true, y_pred)
    print(f"RMSE: {rmse:.2f}, MAE: {mae:.2f}, R: {r2:.2f}")
    return rmse, mae, r2

model = LinearRegression()
model.fit(X_train, y_train)

print("\nDesempenho no Teste:")
y_pred_test = model.predict(X_test)
rmse, mae, r2 = evaluate_model(y_test, y_pred_test)

print("\nCoeficientes do modelo:")
for i, col in enumerate(X.columns):
    print(f"{col}: {model.coef_[i]:.4f}")

def prever_nivel(nivel_ituporanga, chuva_ituporanga, nivel_taio, chuva_taio):
    model = load('modelo_rio_sul.joblib')
    scaler = load('scaler_rio_sul.joblib')
    
    input_data = pd.DataFrame({
        'NivelItuporanga': [nivel_ituporanga],
        'ChuvaItuporanga': [chuva_ituporanga],
        'NivelTaio': [nivel_taio],
        'ChuvaTaio': [chuva_taio]
    })
    
    input_scaled = scaler.transform(input_data)
    return model.predict(input_scaled)[0]

def interface_simples():
    print("\n" + "="*50)
    print("PREVISO DO NVEL DO RIO - RIO DO SUL/SC")

        nt = float(input("Nvel em Tai (cm): "))
        ct = float(input("Chuva em Tai (mm): "))
        
        resultado = prever_nivel(ni, ci, nt, ct)

        print(f"Nvel previsto: {resultado:.2f} cm")

            situacao = "NORMAL"
        elif resultado < 350:
            situacao = "ALTO"
        else:
            situacao = "MUITO ALTO (RISCO)"
            
        print(f"Situao: {situacao}")
        print("="*50)
        
    except Exception as e:
        print(f"\nErro: {str(e)}")
